\subsection{10.3 Charge-Injection}
  Channel charge, $V_{DS} = 0$:
  \begin{equation}
    Q_{ch} = W L C_{ox} V_{eff}
           = W L C_{ox} (V_{GS} - V_t)
  \end{equation}

  Capacitance:
  \begin{equation}
    C = \frac{Q}{V}
  \end{equation}

  Injected voltage, one side of transistor:
  \begin{equation}
    \Delta V = \frac{Q_{ch} / 2}{C}
  \end{equation}
